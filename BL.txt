local SessionStart = os.clock()

local function getSessionTime()
    local elapsed = math.floor(os.clock() - SessionStart)
    local h = math.floor(elapsed / 3600)
    local m = math.floor((elapsed % 3600) / 60)
    local s = elapsed % 60
    if h > 0 then
        return string.format("%dh %02dm %02ds", h, m, s)
    else
        return string.format("%dm %02ds", m, s)
    end
end

local function ensureFolder(path)
    pcall(function()
        if not isfolder(path) then makefolder(path) end
    end)
end

local function saveProfile(name)
    pcall(function()
        ensureFolder("ZebraBoxingLeague")
        ensureFolder(PROFILE_FOLDER)
        local data = {
            RejoinDelay   = AutoRejoin.Delay,
            RejoinMethod  = AutoRejoin.Method,
            RejoinLoad    = AutoRejoin.LoadScript,
        }
        local encoded = game:GetService("HttpService"):JSONEncode(data)
        writefile(PROFILE_FOLDER .. "/" .. name .. ".json", encoded)
    end)
end

local function loadProfile(name)
    local ok = false
    pcall(function()
        local path = PROFILE_FOLDER .. "/" .. name .. ".json"
        if not isfile(path) then return end
        local data = game:GetService("HttpService"):JSONDecode(readfile(path))
        if data.RejoinDelay   then AutoRejoin.Delay            = data.RejoinDelay   end
        if data.RejoinMethod  then AutoRejoin.Method           = data.RejoinMethod  end
        if data.RejoinLoad   ~= nil then AutoRejoin.LoadScript = data.RejoinLoad    end
        ok = true
    end)
    return ok
end

--// HELPER: UI utilities \\--

local function makeRound(parent, radius)
    local c = Instance.new("UICorner", parent)
    c.CornerRadius = UDim.new(0, radius or 8)
    return c
end

local function makeStroke(parent, color, thickness)
    local s = Instance.new("UIStroke", parent)
    s.Color = color or C_STROKE
    s.Thickness = thickness or 1
    return s
end

local function makePadding(parent, t, b, l, r)
    local p = Instance.new("UIPadding", parent)
    p.PaddingTop    = UDim.new(0, t or 0)
    p.PaddingBottom = UDim.new(0, b or 0)
    p.PaddingLeft   = UDim.new(0, l or 0)
    p.PaddingRight  = UDim.new(0, r or 0)
    return p
end

local function getZN()
    for _, gui in ipairs(playerGui:GetChildren()) do
        local zn = gui:FindFirstChild("ZN", true)
        if zn then return zn end
    end
    return nil
end

local function getButtonValue(btn)
    if not btn then return nil end
    local MC = btn:FindFirstChild("MC"); if not MC then return nil end
    local MXVal = MC:FindFirstChild("MX"); if not MXVal then return nil end
    return MXVal.Value
end

local function isButtonDone(btn)
    local val = getButtonValue(btn)
    return val == nil or val <= 0
end

local function allButtonsDone(ZN)
    for i = 1, 4 do
        local btn = ZN:FindFirstChild("B" .. i)
        if btn and not isButtonDone(btn) then return false end
    end
    return true
end

local function getButtonCenter(mc)
    return Vector2.new(mc.AbsolutePosition.X + mc.AbsoluteSize.X/2, mc.AbsolutePosition.Y + mc.AbsoluteSize.Y/2)
end

local function clashButton(btn)
    if not btn or isButtonDone(btn) then return end
    local MC = btn:FindFirstChild("MC"); if not MC then return end
    local MXVal = MC:FindFirstChild("MX"); if not MXVal or MXVal.Value <= 0 then return end
    local center = getButtonCenter(MC)
    pcall(function() VIM:SendMouseMoveEvent(center.X, center.Y, game) end); task.wait()
    pcall(function() VIM:SendMouseButtonEvent(center.X, center.Y, 0, true,  game, 1) end); task.wait()
    pcall(function() VIM:SendMouseButtonEvent(center.X, center.Y, 0, false, game, 1) end); task.wait()
end

local function focusButton(btn)
    if not btn or not ClashSettings.HighlightOnSelect then return end
    local MC = btn:FindFirstChild("MC")
    if MC then pcall(function() GuiService.SelectedObject = MC end) end
end

local function exitUINavigation()
    pcall(function() GuiService.SelectedObject = nil end); task.wait(0.1)
    for _ = 1, 2 do
        pcall(function() VIM:SendKeyEvent(true,  Enum.KeyCode.BackSlash, false, game) end); task.wait(0.05)
        pcall(function() VIM:SendKeyEvent(false, Enum.KeyCode.BackSlash, false, game) end); task.wait(0.05)
    end
    pcall(function() GuiService.SelectedObject = nil end)
end

local autoClashThread = nil
local alreadyExited   = false

local function startAutoClash()
    if autoClashThread then return end
    alreadyExited = false
    autoClashThread = task.spawn(function()
        while ClashSettings.AutoClash do
            local ZN = getZN()
            if ZN then
                if allButtonsDone(ZN) then
                    if not alreadyExited then
                        alreadyExited = true; task.wait(0.3)
                        exitUINavigation()
                        Rayfield:Notify({ Title = "Clash Complete ✅", Content = "All green. Pressed \\ to exit.", Duration = 3, Image = 4483362458 })
                    end
                    task.wait(0.1)
                else
                    alreadyExited = false
                    for i = 1, 4 do
                        if not ClashSettings.AutoClash then break end
                        local btn = ZN:FindFirstChild("B" .. i)
                        if btn and not isButtonDone(btn) then focusButton(btn); clashButton(btn) end
                    end
                end
            else
                task.wait(0.1)
            end
        end
        autoClashThread = nil
    end)
end

local function stopAutoClash()
    ClashSettings.AutoClash = false; autoClashThread = nil
end

local function toggleAutoClash()
    ClashSettings.AutoClash = not ClashSettings.AutoClash
    if ClashSettings.AutoClash then
        startAutoClash()
        Rayfield:Notify({ Title = "Auto Clash ON ✅", Content = "Clicking buttons until green.", Duration = 2, Image = 4483362458 })
    else
        stopAutoClash()
        Rayfield:Notify({ Title = "Auto Clash OFF ❌", Content = "Disabled via keybind.", Duration = 2, Image = 4483362458 })
    end
    updateOverlay()
end

local function freezeCamera()
    if CameraSettings.CurrentlyFrozen then return end
    CameraSettings.CurrentlyFrozen = true
    CameraSettings.FrozenCFrame    = Camera.CFrame
    Camera.CameraType = Enum.CameraType.Scriptable
    CameraSettings.RenderConnection = RunService.RenderStepped:Connect(function()
        if CameraSettings.CurrentlyFrozen and CameraSettings.FrozenCFrame then
            Camera.CameraType = Enum.CameraType.Scriptable
            Camera.CFrame     = CameraSettings.FrozenCFrame
        end
    end)
end

local function unfreezeCamera()
    if not CameraSettings.CurrentlyFrozen then return end
    CameraSettings.CurrentlyFrozen = false
    CameraSettings.FrozenCFrame    = nil
    if CameraSettings.RenderConnection then
        CameraSettings.RenderConnection:Disconnect(); CameraSettings.RenderConnection = nil
    end
    Camera.CameraType = Enum.CameraType.Custom
end

local function isLockedOn()
    local subject = Camera.CameraSubject; if not subject then return false end
    local char = Player.Character; if not char then return false end
    return not (subject:IsDescendantOf(char) or subject == char)
end

local function startLockOnWatcher()
    if CameraSettings.SubjectConnection then return end
    CameraSettings.SubjectConnection = Camera:GetPropertyChangedSignal("CameraSubject"):Connect(function()
        if not CameraSettings.BrokenCamera then return end
        if isLockedOn() then
            freezeCamera()
            Rayfield:Notify({ Title = "Locked On 🎯 — Camera Frozen", Content = "Camera held. Unlock to restore.", Duration = 2, Image = 4483362458 })
        else
            unfreezeCamera()
            Rayfield:Notify({ Title = "Lock Off — Camera Restored 📷", Content = "Camera back to normal.", Duration = 2, Image = 4483362458 })
        end
    end)
end

local function stopLockOnWatcher()
    if CameraSettings.SubjectConnection then
        CameraSettings.SubjectConnection:Disconnect(); CameraSettings.SubjectConnection = nil
    end
    unfreezeCamera()
end

UIS.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if not CameraSettings.BrokenCamera then return end
    if input.KeyCode == CameraSettings.LockOnKeybind then
        if CameraSettings.CurrentlyFrozen then unfreezeCamera() else freezeCamera() end
    end
end)

local function doRejoin()
    local placeId = game.PlaceId
    pcall(function()
        local opts = Instance.new("TeleportOptions")
        if AutoRejoin.LoadScript and SCRIPT_URL ~= "" then
            opts:SetTeleportData({ autoLoad = true, scriptUrl = SCRIPT_URL })
        end
        if AutoRejoin.Method == "NewServer" then opts.ShouldReserveServer = true end
        TeleportService:TeleportAsync(placeId, { Player }, opts)
    end)
end

task.spawn(function()
    pcall(function()
        local joinData = Player:GetJoinData()
        local td = joinData and joinData.TeleportData
        if td and td.autoLoad and td.scriptUrl and td.scriptUrl ~= "" then
            task.wait(3)
            Rayfield:Notify({ Title = "Auto Rejoin — Reloading Script 🔄", Content = "Reloading script after rejoin...", Duration = 4, Image = 4483362458 })
            task.wait(1)
            pcall(function() loadstring(game:HttpGetAsync(td.scriptUrl))() end)
        end
    end)
end)

    end)
end

    if AutoRejoin.Connection then AutoRejoin.Connection:Disconnect(); AutoRejoin.Connection = nil end
end

local reconnectFired = false

local function doReconnect(reason)
    if not AutoReconnect.Enabled then return end
    if reconnectFired then return end
    reconnectFired = true
    warn("[AutoReconnect] Triggered: " .. tostring(reason))
    pcall(function()
        Rayfield:Notify({
            Title   = "Auto Reconnect 🔄",
            Content = "Detected: " .. tostring(reason) .. ". Rejoining in " .. AutoReconnect.Delay .. "s...",
            Duration = AutoReconnect.Delay + 1,
            Image   = 4483362458,
        })
    end)
    task.wait(AutoReconnect.Delay)
    local placeId = game.PlaceId
    pcall(function()
        local opts = Instance.new("TeleportOptions")
        if AutoRejoin.LoadScript and SCRIPT_URL ~= "" then
            opts:SetTeleportData({ autoLoad = true, scriptUrl = SCRIPT_URL })
        end
        TeleportService:TeleportAsync(placeId, { Player }, opts)
    end)
end

local function startAutoReconnect()
    if not AutoReconnect.CoreGuiConnection then
        AutoReconnect.CoreGuiConnection = CoreGui.ChildAdded:Connect(function(child)
            if not AutoReconnect.Enabled then return end
            local n = child.Name:lower()
            if n:find("disconnect") or n:find("error") or n:find("connectionfailed") or n == "errorprompt" then
                task.spawn(doReconnect, "Error screen: " .. child.Name)
            end
        end)
    end
    if not AutoReconnect.NetworkConnection then
        AutoReconnect.NetworkConnection = Players.PlayerRemoving:Connect(function(p)
            if p == Player and AutoReconnect.Enabled then
                task.spawn(doReconnect, "PlayerRemoving fired")
            end
        end)
    end
    if not AutoReconnect.KickConnection then
        AutoReconnect.KickConnection = Player.AncestryChanged:Connect(function(_, parent)
            if parent == nil and AutoReconnect.Enabled then
                task.spawn(doReconnect, "Kicked / disconnected")
            end
        end)
    end
    local lastBeat = os.clock()
    local beatConn = RunService.Heartbeat:Connect(function() lastBeat = os.clock() end)
    task.spawn(function()
        while AutoReconnect.Enabled do
            task.wait(5)
            if AutoReconnect.Enabled and (os.clock() - lastBeat) > 8 then
                beatConn:Disconnect()
                task.spawn(doReconnect, "Heartbeat timeout (network freeze)")
                break
            end
        end
        beatConn:Disconnect()
    end)
end

local function stopAutoReconnect()
    AutoReconnect.Enabled = false
    if AutoReconnect.CoreGuiConnection then AutoReconnect.CoreGuiConnection:Disconnect(); AutoReconnect.CoreGuiConnection = nil end
    if AutoReconnect.NetworkConnection  then AutoReconnect.NetworkConnection:Disconnect();  AutoReconnect.NetworkConnection  = nil end
    if AutoReconnect.KickConnection     then AutoReconnect.KickConnection:Disconnect();     AutoReconnect.KickConnection     = nil end
end

local FpsSnapshot = { Applied = false, DescConnection = nil, Parts = {}, Meshes = {}, Decals = {}, Effects = {}, Lighting = {}, Terrain = {} }

local function takeSnapshot()
    FpsSnapshot.Parts = {}; FpsSnapshot.Meshes = {}; FpsSnapshot.Decals = {}; FpsSnapshot.Effects = {}
    for _, v in next, Workspace:GetDescendants() do
        pcall(function()
            if v:IsA("BasePart") and not v:IsA("MeshPart") then
                table.insert(FpsSnapshot.Parts, { inst=v, CastShadow=v.CastShadow, Material=v.Material, Reflectance=v.Reflectance, MaterialVariant=v.MaterialVariant })
            elseif v:IsA("MeshPart") then
                table.insert(FpsSnapshot.Meshes, { inst=v, CastShadow=v.CastShadow, DoubleSided=v.DoubleSided, RenderFidelity=v.RenderFidelity, TextureID=v.TextureID })
            elseif v:IsA("Decal") or v:IsA("Texture") then
                table.insert(FpsSnapshot.Decals, { inst=v, Transparency=v.Transparency })
            elseif v:IsA("Fire") or v:IsA("SpotLight") or v:IsA("Smoke") or v:IsA("Sparkles") or v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then
                table.insert(FpsSnapshot.Effects, { inst=v, Enabled=v.Enabled })
            end
        end)
    end
    local L = game:GetService("Lighting")
    FpsSnapshot.Lighting = { GlobalShadows=L.GlobalShadows, FogEnd=L.FogEnd, Brightness=L.Brightness }
    local T = Workspace:FindFirstChildOfClass("Terrain")
    if T then FpsSnapshot.Terrain = { WaterReflectance=T.WaterReflectance, WaterTransparency=T.WaterTransparency, WaterWaveSize=T.WaterWaveSize, WaterWaveSpeed=T.WaterWaveSpeed } end
end

local function stripDescendant(v)
    pcall(function()
        if v:IsA("Model") then v.LevelOfDetail="Disabled"; v.ModelStreamingMode="Nonatomic"
        elseif v:IsA("BasePart") and not v:IsA("MeshPart") then v.CastShadow=false; v.Material="Plastic"; v.Reflectance=0; v.MaterialVariant=""
        elseif v:IsA("Decal") or v:IsA("Texture") then v.Transparency=1
        elseif v:IsA("MeshPart") then v.CastShadow=false; v.DoubleSided=false; v.RenderFidelity="Performance"; v.TextureID=10385902758728957
        elseif v:IsA("SpecialMesh") then v.TextureId=0
        elseif v:IsA("ShirtGraphic") then v.Graphic=0
        elseif v:IsA("Shirt") or v:IsA("Pants") then v[v.ClassName.."Template"]=0
        elseif v:IsA("Fire") or v:IsA("SpotLight") or v:IsA("Smoke") or v:IsA("Sparkles") or v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then v.Enabled=false
        elseif v:IsA("Explosion") then v.BlastPressure=1; v.BlastRadius=1
        elseif v:IsA("SurfaceAppearance") or v:IsA("Debris") or v:IsA("MaterialVariant") then v:Destroy()
        elseif v:IsA("Attachment") then v.Visible=false end
    end)
end

local function setNoclip(state)
    if state then
        NoclipSettings.Connection = RunService.Stepped:Connect(function()
            local char = Player.Character
            if not char then return end
            for _, part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") and part.CanCollide then
                    part.CanCollide = false
                end
            end
        end)
    else
        if NoclipSettings.Connection then
            NoclipSettings.Connection:Disconnect()
            NoclipSettings.Connection = nil
        end
        local char = Player.Character
        if char then
            for _, part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") then part.CanCollide = true end
            end
        end
    end
end

    for _, s in ipairs(FpsSnapshot.Parts)   do pcall(function() if s.inst and s.inst.Parent then s.inst.CastShadow=s.CastShadow; s.inst.Material=s.Material; s.inst.Reflectance=s.Reflectance; s.inst.MaterialVariant=s.MaterialVariant end end) end
    for _, s in ipairs(FpsSnapshot.Meshes)  do pcall(function() if s.inst and s.inst.Parent then s.inst.CastShadow=s.CastShadow; s.inst.DoubleSided=s.DoubleSided; s.inst.RenderFidelity=s.RenderFidelity; s.inst.TextureID=s.TextureID end end) end
    for _, s in ipairs(FpsSnapshot.Decals)  do pcall(function() if s.inst and s.inst.Parent then s.inst.Transparency=s.Transparency end end) end
    for _, s in ipairs(FpsSnapshot.Effects) do pcall(function() if s.inst and s.inst.Parent then s.inst.Enabled=s.Enabled end end) end
    pcall(function() local L=game:GetService("Lighting"); sethiddenproperty(L,"Technology",4); L.GlobalShadows=FpsSnapshot.Lighting.GlobalShadows; L.FogEnd=FpsSnapshot.Lighting.FogEnd; L.Brightness=FpsSnapshot.Lighting.Brightness end)
    pcall(function() local T=Workspace:FindFirstChildOfClass("Terrain"); if T and FpsSnapshot.Terrain then sethiddenproperty(T,"Decoration",true); T.WaterReflectance=FpsSnapshot.Terrain.WaterReflectance; T.WaterTransparency=FpsSnapshot.Terrain.WaterTransparency; T.WaterWaveSize=FpsSnapshot.Terrain.WaterWaveSize; T.WaterWaveSpeed=FpsSnapshot.Terrain.WaterWaveSpeed end end)
    FpsSnapshot.Applied = false
    Rayfield:Notify({ Title = "FPS Boost Undone ↩️", Content = "Visuals restored to original state.", Duration = 4, Image = 4483362458 })
end })
FpsTab:CreateSection("Info")
FpsTab:CreateLabel("Snapshots visuals before stripping. Undo restores without rejoining.")
